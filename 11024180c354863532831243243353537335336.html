 /* 2022/01/05 計算BMI 唐英傑 ch11 struct.c */

#include <stdio.h>
#include <stdlib.h>

#define NUM 2 //定義陣列大小 
//宣告結構 
struct student 
{
    char name[8];   
    float Height;
    int weight;
    float BMI; 
};
   
int main()
{
    struct student C_Class[NUM];//宣告結構陣列 
     
    int i;
    int total;
    float Average;
    //輸入要計算的資料 
    printf("Input personal data...\n\n");
    for (Menber=0; Menber<NUM; Menber++) {
       printf("Number #%d :\n",Menber+1);
       
       printf("    C_Class ? ");
       scanf("%s", C_Class[Menber].name);          
       
       printf("    How Height ? ");
       scanf("%f", &C_Class[Menber].Height);
       
       printf("    Weight ? ");
       scanf("%d", &C_Class[Menber].weight);
       
       printf("\n");
    }
    //秀出輸入結果 
    printf("\nDATA LISTING ...\n\n");
    printf("  NUMBER       NAME            Height   WEIGHT     BMI");
    printf("\n\n");
   
    for (Menber = 0; Menber<NUM; Menber++) {
       printf("    #%d",Menber+1);
       printf("    %8s   ",C_Class[Menber].name);
       printf("    %5f %8d",C_Class[Menber].Height,C_Class[Menber].weight);
       C_Class[Menber].BMI=C_Class[Menber].weight/(C_Class[Menber].Height*C_Class[Menber].Height) ;
       printf("    %5f   ",C_Class[Menber].BMI);
       printf("\n");
    }
    //計算BMI總值 
    for (Menber=0,total=0; Menber<NUM; Menber++)
{
   total += C_Class[Menber].BMI;  
}    
//計算BMI平均值
  Average=  total/2;
    printf("\n   Average BMI : %f \n",Average);
   
    system("PAUSE");
    return 0;
}

月儲蓄試算 

/*      簡佑豪   A100D103     2022/01/13      月儲蓄試算                   */
   
    #include <stdio.h>
    #include <stdlib.h>
    
    #define NUM 4 //定義陣列大小 
    //宣告結構 
    struct  monthlysavings
    {
        char name[8];    
        int monthlysalary;
        int rent;
        int mealcost;
        int utilitybill;
        int monthlysavings
    };
    int main()
    {
        struct monthlysavings who[NUM];//宣告結構陣列 
        int i;
        int total;
        float Average;
        //輸入要計算的資料 
        printf("Input personal data...\n\n");
        for (i=0; i<NUM; i++) {
           printf("Number #%d :\n",i+1);
           
           printf("     Who ?  ");
           scanf("%s", who[i].name);            
           
           printf("  monthlysalary  ");
           scanf("%d", &who[i].monthlysalary);
           
           printf("  rent ");
           scanf("%d", &who[i].rent);
            
           printf("  mealcost ");
           scanf("%d", &who[i].mealcost);
           
           printf("  utilitybill  ");
           scanf("%d", &who[i].utilitybill);       
           
           printf("\n");
        }
         printf("\nDATA LISTING ...\n\n");
         printf("  NUMBER       NAME     monthlysalary  rent  mealcost         ");
         printf("\n\n");
        
        for (i = 0; i<NUM; i++) {
         printf("    #%d",i+1);
         printf("    %8s   ",who[i].name);
         printf("    %5d %8d",who[i].monthlysalary,who[i].rent);
          printf("    %5d %8d",who[i].mealcost,who[i].utilitybill);
         who[i].monthlysavings=who[i].monthlysalary-(who[i].rent+who[i].mealcost+who[i].utilitybill) ;
         printf("    %5d   ",who[i].monthlysavings);
         printf("\n");
        }
        
    for (i=0,total=0; i<NUM; i++)
     {
            total += who[i].monthlysavings;     
     }    
     
      Average=  total/2;
        printf("\n   Average monthlysavings : %f \n",Average);
        
        system("PAUSE");
        return 0;
    }

/************************************************************************/
/*    作業迴圈使用               陳冠良  A100E106  2021/11/24                     */
#include <stdio.h>

int main()
{
    int YearMain=2021;
    int UaryMain=12;
    int DayMain=30;
    //for 迴圈遞增型用法
    for(int Year = 2015; Year<=YearMain ; Year=Year+1)
    {
        printf("\n--------------------------");
        printf("\n%d年",Year);
        
        for(int Uary=1;Uary<=UaryMain;Uary=Uary+1)
        {
            
            if(Uary==2)//如果等於2月日期等於28天
            {
                DayMain = 28;
            }else
            {
                if(Uary==1 || Uary==3 || Uary==5 || Uary==7 || Uary==8 || Uary==10 || Uary==12)
                DayMain = 31;
                if(Uary==4 || Uary==6 || Uary==9 || Uary==11)
                DayMain = 30;
            }
            if(Uary<=UaryMain)
            {
                printf("\n");//getchar();停止等待輸入
                printf("--------------------------\n");
            }
            printf("  %d月\n",Uary);
            printf("--------------------------");
            //getchar();
           for(int Day=1;Day<=DayMain;Day=Day+1)
           {
               if(Day/7*7+1==Day)//先除7的商在乘7在+1 = 7的倍數+1

               }
               printf("%d ",Day);
               
           }
        }
    }
    return 0;
}
